# Episteme — Generator (generator.py)
# Purpose: Generate answers + follow-up + citations + summary

import os
from dotenv import load_dotenv
from langchain_groq import ChatGroq
from langchain_core.prompts import ChatPromptTemplate
from retriever import hybrid_retrieve

# ─── Load Environment Variables ───────────────────────────────
load_dotenv()

# ─── Constants ────────────────────────────────────────────────
GROQ_MODEL  = "llama-3.3-70b-versatile"
TEMPERATURE = 0.1

# ─── System Prompt ────────────────────────────────────────────
SYSTEM_PROMPT = """You are Episteme, an intelligent STEM knowledge assistant.

CRITICAL LANGUAGE RULES:
1. Detect user's language and respond in SAME language only
2. English → English only
3. Roman Urdu → Roman Urdu only (NEVER Hindi/Devanagari)
4. Any other language → same language

ROMAN URDU DETECTION:
Words: kya, hai, kaise, karta, hota, batao, aur, mein, nahi, toh
→ Respond in Roman Urdu ONLY — never mix Hindi

CONTENT RULES:
1. Answer from provided context first
2. PDF content → prioritize in answer
3. If context insufficient → use knowledge, mention it
4. ONLY use directly relevant ArXiv papers
5. For math/engineering → show LaTeX formulas where helpful
6. Structure: Definition → How it works → Example → Applications

RESPONSE FORMAT — always end with this EXACT block:
---CONFIDENCE---
LEVEL: [HIGH/MEDIUM/LOW]
REASON: [one line why]
---FOLLOWUP---
1. [First follow-up question in same language as user]
2. [Second follow-up question]
3. [Third follow-up question]
---END---

CONVERSATION HISTORY:
{chat_history}

CONTEXT FROM SOURCES:
{context}"""

# ─── Summary Prompt ───────────────────────────────────────────
SUMMARY_PROMPT = """You are Episteme. Create a professional session summary.

Conversation:
{conversation}

Provide a structured summary with these sections:

## Session Summary

**Topics Discussed:**
- List main topics

**Key Concepts Learned:**
- Important concepts

**Questions Asked:**
- Main questions

**Key Takeaways:**
- Most important points

**Recommended Next Steps:**
- What to study next

Keep it concise and professional."""


# ─── Initialize LLM ───────────────────────────────────────────
def load_llm():
    """Load Groq LLM."""
    return ChatGroq(
        model       = GROQ_MODEL,
        temperature = TEMPERATURE,
        api_key     = os.getenv("GROQ_API_KEY"),
    )


# ─── Format Context ───────────────────────────────────────────
def format_context(results: dict) -> str:
    """Format retrieved chunks into context string."""
    context_parts = []
    for doc in results["all"]:
        doc_type = doc.metadata.get("type", "unknown")
        source   = doc.metadata.get("source", "Unknown")
        title    = doc.metadata.get("title", "")

        if doc_type == "pdf":
            label = f"📄 PDF: {title}"
        elif doc_type == "wikipedia_live":
            label = f"📚 Wikipedia: {title}"
        elif doc_type == "arxiv_live":
            label = f"🔬 ArXiv: {title}"
        elif doc_type == "pubmed":
            label = f"🔬 PubMed: {title}"
        else:
            label = f"📖 Knowledge Base: {title}"

        context_parts.append(
            f"[{label}]\nSource: {source}\n{doc.page_content}"
        )
    return "\n\n---\n\n".join(context_parts)


# ─── Format Chat History ──────────────────────────────────────
def format_chat_history(history: list) -> str:
    """Format conversation history."""
    if not history:
        return "No previous conversation."
    formatted = []
    for msg in history[-4:]:
        role    = "User" if msg["role"] == "user" else "Episteme"
        content = msg["content"][:200]
        formatted.append(f"{role}: {content}")
    return "\n".join(formatted)


# ─── Parse Response ───────────────────────────────────────────
def parse_response(text: str) -> dict:
    """Extract answer, confidence, and follow-ups from response."""
    answer     = text
    confidence = "HIGH"
    reason     = ""
    followups  = []

    # Extract confidence
    if "---CONFIDENCE---" in text:
        parts  = text.split("---CONFIDENCE---")
        answer = parts[0].strip()
        rest   = parts[1] if len(parts) > 1 else ""

        for line in rest.split("\n"):
            if line.startswith("LEVEL:"):
                confidence = line.replace("LEVEL:", "").strip()
            elif line.startswith("REASON:"):
                reason = line.replace("REASON:", "").strip()

    # Extract follow-ups
    if "---FOLLOWUP---" in text:
        fq_part = text.split("---FOLLOWUP---")[-1]
        fq_part = fq_part.split("---END---")[0].strip()
        for line in fq_part.split("\n"):
            line = line.strip()
            if line and line[0].isdigit():
                q = line.split(".", 1)[-1].strip()
                if q:
                    followups.append(q)

    # Clean answer
    for marker in ["---CONFIDENCE---", "---FOLLOWUP---", "---END---"]:
        answer = answer.split(marker)[0]
    answer = answer.strip()

    return {
        "answer"     : answer,
        "confidence" : confidence,
        "reason"     : reason,
        "followups"  : followups[:3],
    }


# ─── Generate IEEE Citation ───────────────────────────────────
def generate_citation(sources: list) -> str:
    """Generate IEEE format citations."""
    citations = []
    for i, src in enumerate(sources[:3], 1):
        if "wikipedia" in src.lower():
            topic = src.split("/")[-1].replace("_", " ")
            citations.append(
                f"[{i}] Wikipedia, \"{topic},\" [Online]. Available: {src}"
            )
        elif "arxiv" in src.lower():
            citations.append(
                f"[{i}] ArXiv Research Paper, [Online]. Available: {src}"
            )
        elif "pubmed" in src.lower():
            citations.append(
                f"[{i}] PubMed Research, [Online]. Available: {src}"
            )
        else:
            citations.append(f"[{i}] {src}")
    return "\n".join(citations)


# ─── Generate Answer ──────────────────────────────────────────
def generate_answer(
    query         : str,
    chat_history  : list = [],
    uploaded_pdfs : list = [],
) -> dict:
    """Full pipeline: Query → Retrieve → Generate → Answer"""

    # Step 1: Retrieve
    results = hybrid_retrieve(
        query         = query,
        chat_history  = chat_history,
        uploaded_pdfs = uploaded_pdfs,
    )
    context = format_context(results)
    history = format_chat_history(chat_history)

    # Step 2: Prompt
    prompt = ChatPromptTemplate.from_messages([
        ("system", SYSTEM_PROMPT),
        ("human", "{question}"),
    ])

    # Step 3: Generate
    llm    = load_llm()
    chain  = prompt | llm
    result = chain.invoke({
        "context"      : context,
        "chat_history" : history,
        "question"     : query,
    })

    # Step 4: Parse
    parsed = parse_response(result.content)

    # Step 5: Sources
    raw_sources = []
    for doc in results["all"]:
        src   = doc.metadata.get("source", "Unknown")
        dtype = doc.metadata.get("type", "")
        if dtype == "pdf":
            raw_sources.append(f"📄 {src}")
        elif dtype == "wikipedia_live":
            raw_sources.append(f"📚 {src}")
        elif dtype in ["arxiv_live", "arxiv"]:
            raw_sources.append(f"🔬 {src}")
        elif dtype == "pubmed":
            raw_sources.append(f"🔬 {src}")
        else:
            raw_sources.append(f"📖 {src}")

    unique_sources = list(set(raw_sources))
    clean_sources  = [s.split(" ", 1)[-1] for s in unique_sources]
    citations      = generate_citation(clean_sources)

    return {
        "answer"      : parsed["answer"],
        "confidence"  : parsed["confidence"],
        "reason"      : parsed["reason"],
        "followups"   : parsed["followups"],
        "citations"   : citations,
        "sources"     : unique_sources,
        "chunks"      : len(results["all"]),
        "is_research" : results["is_research"],
        "pdf_used"    : len(results["pdf"]) > 0,
    }


# ─── Generate Session Summary ─────────────────────────────────
def generate_summary(messages: list) -> str:
    """Generate session summary from chat history."""
    if not messages:
        return "No conversation to summarize."

    conversation = ""
    for msg in messages:
        role    = "User" if msg["role"] == "user" else "Episteme"
        content = msg["content"][:500]
        conversation += f"{role}: {content}\n\n"

    prompt = ChatPromptTemplate.from_messages([
        ("human", SUMMARY_PROMPT),
    ])

    llm    = load_llm()
    chain  = prompt | llm
    result = chain.invoke({"conversation": conversation})
    return result.content
