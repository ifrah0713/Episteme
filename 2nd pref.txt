# ─── Helpers ──────────────────────────────────────────────────
def handle_pdf_upload(files: list):
    remaining = MAX_PDFS - len(st.session_state.uploaded_pdfs)
    for f in files[:remaining]:
        if f.name not in st.session_state.uploaded_pdfs:
            with st.spinner(f"Processing {f.name}..."):
                result = process_pdf(f)
            if result["success"]:
                st.session_state.uploaded_pdfs.append(f.name)
                st.markdown(
                    f'<div class="pdf-success">✅ {f.name} — {result["pages"]} pages</div>',
                    unsafe_allow_html=True,
                )
            else:
                st.markdown(
                    f'<div class="pdf-error">❌ {result["error"]}</div>',
                    unsafe_allow_html=True,
                )


def create_summary_pdf(summary_text: str, user_name: str) -> bytes:
    buffer = io.BytesIO()
    doc    = SimpleDocTemplate(buffer, pagesize=letter,
                               rightMargin=72, leftMargin=72,
                               topMargin=72, bottomMargin=72)
    styles = getSampleStyleSheet()
    story  = []
    story.append(Paragraph("Episteme — Session Summary",
        ParagraphStyle("T", parent=styles["Heading1"], fontSize=22,
                       textColor=colors.HexColor("#10a37f"), spaceAfter=6)))
    now = datetime.now().strftime("%B %d, %Y at %I:%M %p")
    story.append(Paragraph(f"User: {user_name} | {now}",
        ParagraphStyle("M", parent=styles["Normal"], fontSize=10,
                       textColor=colors.gray, spaceAfter=20)))
    story.append(Spacer(1, 0.2*inch))
    body = ParagraphStyle("B", parent=styles["Normal"],
                          fontSize=11, leading=16, spaceAfter=8)
    for line in summary_text.split("\n"):
        line = line.strip()
        if not line:
            story.append(Spacer(1, 0.1*inch))
        elif line.startswith("##"):
            story.append(Paragraph(line.replace("##","").strip(),
                ParagraphStyle("H2", parent=styles["Heading2"], fontSize=14,
                               textColor=colors.HexColor("#10a37f"), spaceAfter=6)))
        elif line.startswith("-"):
            story.append(Paragraph(f"• {line[1:].strip()}", body))
        else:
            story.append(Paragraph(line.replace("**",""), body))
    doc.build(story)
    return buffer.getvalue()


def get_confidence_html(confidence: str, reason: str) -> str:
    if confidence == "HIGH":
        return '<span class="conf-high">🟢 High Confidence</span>'
    elif confidence == "MEDIUM":
        return f'<span class="conf-medium">🟡 Medium — {reason}</span>'
    else:
        return '<span class="conf-low">🔴 Low Confidence — please verify</span>'

            followups   = response.get("followups", [])
            citations   = response.get("citations", "")
            confidence  = response.get("confidence", "HIGH")
            reason      = response.get("reason", "")
            is_research = response["is_research"]
            pdf_used    = response.get("pdf_used", False)

        # ── Answer ────────────────────────────────────────────
        st.markdown(answer)

        # ── Confidence + Badges ───────────────────────────────
        badges = get_confidence_html(confidence, reason)
        if is_research:
            badges += '<span class="badge badge-arxiv">🔬 ArXiv</span>'
        if pdf_used:
            badges += f'<span class="badge badge-pdf">📄 PDF</span>'
        st.markdown(badges, unsafe_allow_html=True)

        # ── Sources — NO expander ─────────────────────────────
        if sources:
            st.markdown(
                f"<div style='font-size:0.72rem;color:{TEXT_SUB};"
                f"margin-top:12px;margin-bottom:5px;font-weight:600;"
                f"text-transform:uppercase;letter-spacing:0.06em;'>"
                f"📚 Sources</div>",
                unsafe_allow_html=True,
            )
            for src in sources[:4]:
                st.markdown(
                    f'<div class="source-box">{src}</div>',
                    unsafe_allow_html=True,
                )

        # ── Citations — NO expander ───────────────────────────
        if citations:
            st.markdown(
                f"<div style='font-size:0.72rem;color:{TEXT_SUB};"
                f"margin-top:12px;margin-bottom:5px;font-weight:600;"
                f"text-transform:uppercase;letter-spacing:0.06em;'>"
                f"📝 IEEE Citations</div>",
                unsafe_allow_html=True,
            )
            st.markdown(
                f'<div class="citation-box">'
                f'{citations.replace(chr(10), "<br>")}'
                f'</div>',
                unsafe_allow_html=True,
            )

        # ── Follow-ups ────────────────────────────────────────
        if followups:
            st.markdown(
                '<div class="followup-label">💡 You might also ask</div>',
                unsafe_allow_html=True,
            )
            for i, fq in enumerate(followups):
                if st.button(
                    f"→ {fq}",
                    key                 = f"fq_{st.session_state.chat_id}_{i}_{len(st.session_state.messages)}",
                    use_container_width = True,
                ):
                    st.session_state.messages.append({"role": "user", "content": fq})
                    save_message(st.session_state.chat_id, "user", fq)
                    st.rerun()

        # ── Save ──────────────────────────────────────────────
        full_response = f"{answer}\n\n**Sources:** {', '.join(sources[:3])}"
        save_message(st.session_state.chat_id, "assistant", full_response)
        st.session_state.messages.append({
            "role"    : "assistant",
            "content" : full_response,
        })
